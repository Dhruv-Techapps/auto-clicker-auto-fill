  filePath={getDocsRelativePath('/static/docs/[version]/assets/html/input2.html')}
  variables={{
    element_finder: '//html',

    value2: 'ScrollTo::XPath'
  }}
/>

<BsTable>

| **ScrollTo::**                              | **Description**             |
| ------------------------------------------- | --------------------------- |
| `ScrollTo::XPath`                           | Scroll page to the element. |
| `ScrollTo::Top` / `ScrollTo::TopLeft`       | Scroll to Top Left.         |
| `ScrollTo::TopRight`                        | Scroll to Top Right.        |
| `ScrollTo::Bottom` / `ScrollTo::BottomLeft` | Scroll to Bottom Left.      |
| `ScrollTo::BottomRight`                     | Scroll to Bottom Right.     |

</BsTable>
